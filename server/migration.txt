# crear python env
python -m venv .venv
.venv\Scripts\activate

# para crear el archivo de requerimientos y actualizar librerias

pip install -r requirements.txt #instalamos las dependencias

.venv\Scripts\activate #activamos venv
python -m uvicorn index:server --reload --port 3001 #inicializamos, index:server es el sv nombre + archivo nombre

pip install fastapi uvicorn websockets
